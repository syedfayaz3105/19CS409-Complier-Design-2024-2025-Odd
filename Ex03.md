#include <stdio.h>

// Declare yylex and yyerror before their usage
int yylex(void);
void yyerror(char *s);

/* This YACC program is for recognizing the Expression */
%}

%token ID PLUS MINUS MULTIPLICATION DIVISION

%% 

statement: 
    ID '=' E {
        printf("\nValid arithmetic expression\n");
    }
    ;

E: 
    E PLUS ID
    | E MINUS ID
    | E MULTIPLICATION ID
    | E DIVISION ID
    | ID
    ;

%% 

extern FILE* yyin; 

int main() {
    do {
        yyparse();
    } while (!feof(yyin)); 
    return 0; 
}
